\label{keys_8c_8caf96f2cb44c74717d932fb6b5fa611}


Gets the state of the key given by index. 

\begin{Desc}
\item[Parameters:]
\begin{description}
\item[{\em key\_\-index}]Index number of the key. \end{description}
\end{Desc}
\begin{Desc}
\item[Returns:]One of the following values will be returned as long as the given key is pressed/not pressed: \begin{TabularC}{3}
\hline
{\bf Value}&{\bf Description} \\\hline
\hyperlink{keys_8h_42819d478612ef90666b18dfacd44ea6}{KEY\_\-NOT\_\-PRESSED} &if the key is currently not pressed,&\\\hline
\hyperlink{keys_8h_8386b89daf8ec3a1805554fd95b80a9e}{KEY\_\-PRESSED} &key has been pressed for $<$ 1 sec, \\\hline
\hyperlink{keys_8h_8264ff33541be69d92196213278ddb81}{KEY\_\-PRESSED\_\-FOR\_\-1S} &if the key has been pressed for 1..2 sec, \\\hline
\hyperlink{keys_8h_fdaced0342c2f56dec53f5c812099e34}{KEY\_\-PRESSED\_\-FOR\_\-2S} &if the key has been pressed for 2..3 sec, \\\hline
\hyperlink{keys_8h_075d288eef80f71635ec499fca34b3f9}{KEY\_\-PRESSED\_\-FOR\_\-3S} &if the key has been pressed for 3..4 sec, \\\hline
\hyperlink{keys_8h_04389585c80c4c5fee943cb48fe308e5}{KEY\_\-PRESSED\_\-FOR\_\-4S} &if the key has been pressed for 4..5 sec, \\\hline
\hyperlink{keys_8h_43ef75877ab3ce6a0b268cfbe22d0360}{KEY\_\-PRESSED\_\-FOR\_\-5S} &if the key has been pressed for $>$ 5 sec, \\\hline
\end{TabularC}
There is a special behaviour {\em once\/} after key release. The function will then return one of these values: \begin{TabularC}{3}
\hline
{\bf Value}&{\bf Description} \\\hline
\hyperlink{keys_8h_7289f1ae18cb4742bf9f4e1965b63071}{KEY\_\-REL\_\-AFTER\_\-10MS} &if the key was released after $<$ 1 sec,&\\\hline
\hyperlink{keys_8h_9e9346747ee34398968a604b6a88f08e}{KEY\_\-REL\_\-AFTER\_\-1S} &if the key was released after 1..2 sec, \\\hline
\hyperlink{keys_8h_28bca403d46afffd63c71bc9b977dc8b}{KEY\_\-REL\_\-AFTER\_\-2S} &if the key was released after 2..3 sec, \\\hline
\hyperlink{keys_8h_69d7b969c4fc42011d96809a89bf1fb0}{KEY\_\-REL\_\-AFTER\_\-3S} &if the key was released after 3..4 sec, \\\hline
\hyperlink{keys_8h_06307b2b74026911d5fa8c789ed081ef}{KEY\_\-REL\_\-AFTER\_\-4S} &if the key was released after 4..5 sec, \\\hline
\hyperlink{keys_8h_b2aa33ea89b96412698f3e087dba5596}{KEY\_\-REL\_\-AFTER\_\-5S} &if the key was released after $>$ 5 sec, \\\hline
\end{TabularC}
\end{Desc}
\begin{Desc}
\item[Note:]To find out if the key is currently pressed this sentence must be true: (Get\-Key\-State(mykey) \& KEY\_\-PRESSED) == KEY\_\-PRESSED \end{Desc}
